const r1 = await fetch('/api/auth/request-token', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    apiKey, encKey, userId,
    userPassword, dob, appSource
  })
});
const { body: { RequestToken } } = await r1.json();

// ─── Step 2: Enter TOTP → get JWT ───
const r2 = await fetch('/api/auth/get-jwt', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    apiKey, encKey, userId,
    requestToken: RequestToken, totp
  })
});
const { body: { JWTToken } } = await r2.json();
